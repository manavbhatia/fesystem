00589 {
00590   static double ref_temp = this->analysis_discipline.getFESystemController().
00591   analysis_case->getRealParameter("REFERENCE_TEMP");

00600     {
00601     nodal_temp.resize(n_nodes);
00602     nodal_temp_sens.resize(n_nodes);

00605   nodal_temp.zero(); nodal_temp_sens.zero();
00606   
00607   // get the loads for this element
00608   this->extractNodalTemperatureVectorFromLoads(nodal_temp, false);

00618     this->extractNodalTemperatureVectorFromLoads(nodal_temp_sens, true);

00620       {

00622         this->elem_property_card->getFactorSensitivityForGlobalParameter

00624

00646   switch (sensitivity_calculation)
00647     {

00650       {
00651       for (unsigned int i=0; i < n_nodes; i++)

00657     
00658     case true:
00659       {
00660         switch (this->sensitivity_parameter)
00661           {
00662           case PROPERTY_PARAMETER_ENUM_ID:
00663             {

00673                 {

00693                                        design_point_enum_ID,
00694                                        FESystemElem::ELEM_VOLUME::num(), LAGRANGE);
00695               this->getFESystemElemQty(FESystemElem::N_Y_N_FACTOR::num(), &ny_n_matrix,
00696                                        design_point_enum_ID,
00697                                        FESystemElem::ELEM_VOLUME::num(), LAGRANGE);

00704                                                        FESystemElem::ELEM_VOLUME::num(),

00711                 for (unsigned int j=0; j<n_nodes; j++)

00714                                    nx_n_matrix(i,j) * nodal_temp_sens(j)) +

00716                   sum2 += factor* ( ny_n_matrix_sens(i,j) * (nodal_temp(j)-ref_temp) +

00739                                  design_point_enum_ID,
00740                                  FESystemElem::ELEM_VOLUME::num(), LAGRANGE);

00747             {

00749             sum2 += ny_n_matrix(i,j) * (factor* (nodal_temp(j)-ref_temp));
00750             }
00751           (*vector)(i) = sum1;
00752           (*vector)(n_nodes + i) = sum2;
00753           }
00754       }
00755       break;
00756     }
00757 }
