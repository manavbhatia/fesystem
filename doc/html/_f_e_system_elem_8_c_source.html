01351       this->initialize_element(design_point_enum_ID);

01355     {
01356     case 3:
01357       {
01358         // keep going
01359       }
01360       break;
01361       
01362     default:
01363       vector->resize(3);
01364       break;
01365     }
01366   
01367         vector->zero();
01368   
01369   static bool load_on_elem_face;
01370   load_on_elem_face = false;
01371     
01372   unsigned int side_num = 
01373     this->getSideNumberFromDomainEnum(domain);

01382          ExcInternalError());
01383

01385     {
01386     case 0:
01387       // then the load is on the element face of a quad element
01388       load_on_elem_face = true;

01390

01393         // nothing to be done, since the load is not on elem face

01401       {

01411           {
01412           case 0:
01413             {
01414               // scale the vector by -1.0

01419           case 1:
01420             break;

01426       }
01427       break;

01430       {
01431         switch (load_on_elem_face)
01432           {
01433           case true:
01434             {
01435               // calculate the vector along 0->1 and 0->2. The cross product of these two will give the
01436               // surface normal
01437               const Point& point0 = local_elem->point(0);
01438               const Point& point1 = local_elem->point(1);
01439               const Point& point2 = local_elem->point(2);
01440               
01441               //calculate the position vectors from point 0

01443               vector01.assign(point1 - point0);
01444               vector02.assign(point2 - point0);
01445               
01446               
01447               // calculate the cross product and scale it by the side (for +ve or -ve)
01448               tmp_vector.assign(vector01.cross(vector02));
01449               
01450               normal.assign(tmp_vector.unit());
01451             }
01452             break;
01453             
01454           case false:
01455           default:
01456             {
01457               // calculate the vector along the length of this side
01458               // the surface normal will be the cross-product of this vector with the z-vector
01459               // calculate the vector along the length of the element from 0->1
01460               std::auto_ptr<Elem> side_elem(local_elem->build_side(side_num).release());
01461               
01462               const Point& point0 = side_elem->point(0);
01463               const Point& point1 = side_elem->point(1);
01464               
01465               static Point vector01, z_vector, tmp_vector;
01466               z_vector.zero();
01467               z_vector(2) =  1.0;
01468               vector01.assign(point1 - point0);
01469               tmp_vector.assign(vector01.cross(z_vector));
01470               normal.assign(tmp_vector.unit());
01471             }
01472             break;
01473           }
01474       }
01475       break;
01476       
01477     case 3:
01478       {
01479         // get the side of this element, and follow the same procedure as the one for load_on_face
01480         std::auto_ptr<Elem> side_elem(local_elem->build_side(side_num).release());
01481         
01482         const Point& point0 = side_elem->point(0);
01483         const Point& point1 = side_elem->point(1);
01484         const Point& point2 = side_elem->point(2);
01485         
01486         //calculate the position vectors from point 0
01487         static Point vector01, vector02, tmp_vector;
01488         vector01.assign(point1 - point0);
01489         vector02.assign(point2 - point0);
01490         
01491         
01492         // calculate the cross product and scale it by the side (for +ve or -ve)
01493         tmp_vector.assign(vector01.cross(vector02));
01494         
01495         normal.assign(tmp_vector.unit());
01496       }
01497       break;
01498     }
01499   
01500   // now copy the normal vector components
01501   for (unsigned int i=0; i<3; i++)
01502     (*vector)(i)  = normal(i);
01503 }

01752 double FESystemElem::FESystemElemBase::getElementMass(bool sensitivity)
01753 {
01754   // then, check about the property card
01755   if (!this->property_card_initialized)
01756     this->initPropertyCard();
01757   
01758   double factor = 0.0, factor_sens = 0.0, vol = 0.0;
01759   DenseVector<double> vector;
01760   
01761   // get the mass factor from the element data card
01762   switch (this->dimension)
01763   {
01764     case 1:
01765     case 3:
01766       this->elem_property_card->getFactor(factor, MASS_FACTOR::num());
01767       break;
01768       
01769     case 2:
01770       this->elem_property_card->getFactor(factor, MEMBRANE_MASS_FACTOR::num());
01771       break;
01772       
01773     default:
01774       ExcInternalError();
01775   }
01776 
01777   // get the value of volume
01778   switch (this->dimension)
01779   {
01780     case 1:
01781     {
01782       static Elem* elem;
01783       elem = this->geometric_elems_for_DV_map[FESystemElem::BASE_ELEM::num()];
01784       
01785       // calculate the length of the element
01786       const Point& point0 = elem->point(0);
01787       const Point& point1 = elem->point(1);
01788       
01789       static double dx, dy, dz;
01790       dx = point0(0)-point1(0);
01791       dy = point0(1)-point1(1);
01792       dz = point0(2)-point1(2);
01793       vol = pow((dx*dx+dy*dy+dz*dz),0.5);
01794     }
01795       break;
01796       
01797     case 2:
01798     case 3:
01799     {
01800       this->getFESystemElemQty(FESystemElem::N_FACTOR::num(), &vector, 
01801                                FESystemElem::BASE_ELEM::num(),
01802                                FESystemElem::ELEM_VOLUME::num(), LAGRANGE);
01803       for (unsigned int i=0; i < vector.size(); i++)
01804         vol += vector.el(i);
01805     }
01806       
01807     default:
01808       ExcInternalError();
01809   }
01810 
01811   // check for property sensitivity
01812   if (sensitivity && 
01813       this->sensitivity_parameter == DesignData::PROPERTY_PARAMETER::num())
01814     {
01815       if (!this->elem_property_card->checkElemAndMaterialCardGlobalParameterDependence
01816           (this->sensitivity_parameter_ID))
01817         return 0.0;
01818       
01819       switch (this->dimension)
01820       {
01821         case 1:
01822         case 3:
01823           this->elem_property_card->getFactorSensitivityForGlobalParameter
01824           (factor_sens, MASS_FACTOR::num(), this->sensitivity_parameter_ID);
01825           break;
01826           
01827         case 2:
01828           this->elem_property_card->getFactorSensitivityForGlobalParameter
01829           (factor_sens, MEMBRANE_MASS_FACTOR::num(), this->sensitivity_parameter_ID);
01830           break;
01831           
01832         default:
01833           ExcInternalError();
01834       }
01835     }
01836   
01837   // to get the jacobian factor, get the N_FACTOR integrated over the volume
01838   // and then add all the elements of the vector
01839   if (sensitivity)
01840     {
01841       switch (this->sensitivity_parameter)
01842       {
01843         case PROPERTY_PARAMETER_ENUM_ID:
01844           vol *= factor_sens;
01845           break;
01846           
01847         case SHAPE_PARAMETER_ENUM_ID:
01848         {
01849           // this will set the sensitivity value of volume of element
01850           switch (this->dimension)
01851           {
01852             case 1:
01853             {
01854               static Elem* perturbed_elem;
01855               perturbed_elem = 
01856               this->geometric_elems_for_DV_map[FESystemElem::BASE_PLUS_DELTA_ELEM::num()];
01857               // calculate the length for the perturbed element, and find the sensitivity
01858               // calculate the length of the element
01859               const Point& point0 = perturbed_elem->point(0);
01860               const Point& point1 = perturbed_elem->point(1);
01861               
01862               static double dx, dy, dz;
01863               dx = point0(0)-point1(0);
01864               dy = point0(1)-point1(1);
01865               dz = point0(2)-point1(2);
01866               vol -= pow((dx*dx+dy*dy+dz*dz),0.5);
01867               vol /= -0.5;
01868             }
