01056 
01057 template <class QtyType >
01058 void 
01059 FESystemElem::FESystemElemBase::getFESystemElemQtyShapeSensitivity(const unsigned int name,
01060                                                                    QtyType* qty,
01061                                                                    const unsigned int domain,
01062                                                                    const FEFamily& family)
01063 {
01064   Assert(this->elem_is_initialized_for_DV_map[FESystemElem::BASE_PLUS_DELTA_ELEM::num()],
01065          ExcInvalidState());
01066   assert (qty != NULL);
01067   
01068   // create quantities for the perturbed and base elements
01069   static QtyType  base_qty;
01070   
01071   // get the string name for the quantity
01072   static std::string qty_string_name, qty_sensitivity_string_name;
01073   qty_string_name.clear();  qty_sensitivity_string_name.clear();
01074   
01075   qty_string_name = this->getStringNameForFESystemElemQty(name, 
01076                                                           FESystemElem::BASE_ELEM::num(),
01077                                                           domain, family);
01078   qty_sensitivity_string_name = 
01079     this->getStringNameForFESystemElemQtySensitivity(name,
01080                                                      this->sensitivity_parameter_ID,
01081                                                      domain,
01082                                                      family);
01083   
01084   // check if the quantity exists in the element database or not
01085   // if not, calculate and store it
01086   if (this->elem_data_storage->checkIfQtyExists
01087       (this->discipline_enumID, this->elem_ID, qty_sensitivity_string_name))
01088     {
01089     // get the factor from the database
01090     this->elem_data_storage->get_element_data(*qty, 
01091                                               this->discipline_enumID, this->elem_ID, 
01092                                               qty_sensitivity_string_name);
01093     }
01094   else
01095     {
01096     // next, obtain the quantity for the perturbed element
01097     this->calculateFESystemElemQty(qty, name,
01098                                    FESystemElem::BASE_PLUS_DELTA_ELEM::num(), 
01099                                    domain, family);
01100     
01101     // also, obtain the quantity for the base element. The base quantity also needs to
01102     // be evaluated at the same load case, dof_value for sensitivity calculation
01103     base_qty.zero();
01104     this->getFESystemElemQty(name, &base_qty,  FESystemElem::BASE_ELEM::num(),
01105                              domain, family);
01106     
01107     qty->add(-1.0, base_qty);
01108     qty->scale(1.0/this->perturbation);
01109     
01110     this->elem_data_storage->add_element_data(*qty,
01111                                               this->discipline_enumID, this->elem_ID, 
01112                                               qty_sensitivity_string_name);
01113     }
01114 }

01132   Assert(data_card != NULL, ExcEmptyObject());
01133

01141   this->elem_property_card = data_card;
01142

01144   if (!this->fe_quadrature_data_initialized)
01145     this->initializeFEAndQuadratureDataStructures();
01146 
01147   
01148   // also, resize the dof vectors if that has not already been done
01149   static const unsigned int n_dofs = this->getNDofs();
01150   
01151   if (this->dof_values_vec.size() != (this->getTransientSystemOrder()+1))
01152     {
01153     this->dof_values_vec.clear();
01154     this->dof_value_sensitivity_vec.clear();
01155     
01156     for (unsigned int i=0; i <= this->getTransientSystemOrder(); i++)
01157       {
01158       this->dof_values_vec.push_back(DenseVector<double>(n_dofs));
01159       this->dof_values_vec[i].zero();
01160       this->dof_value_sensitivity_vec.push_back(DenseVector<double>(n_dofs));
01161       this->dof_value_sensitivity_vec[i].zero();
01162       }
01163     }

01231 inline
01232 void 
01233 FESystemElem::FESystemElemBase::deleteElemLocalData(const unsigned int design_point_enum_ID)
01234 {
01235   unsigned int n_nodes;
01236   
01237   // clear the base element data
01238   // first release the local elem pointer in case it is 3D, since for that case
01239   //  the local element is just a pointer to the base elem
01240   
01241   switch (this->local_elem_for_DV_map[design_point_enum_ID] == NULL)
01242     {
01243     case true:
01244       {
01245         // nothing to be done here, since the element was not created.
01246       }
01247       break;
01248       
01249     case false:
01250     default:
01251       {
01252         switch (this->local_elem_for_DV_map[design_point_enum_ID]->dim())
01253           {
01254           case 3:
01255             this->local_elem_for_DV_map[design_point_enum_ID] = NULL; 
01256             break;
01257             
01258           default:
01259             {
01260               // delete the element
01261               n_nodes = this->local_elem_for_DV_map[design_point_enum_ID]->n_nodes();
01262               delete this->local_elem_for_DV_map[design_point_enum_ID];
01263               this->local_elem_for_DV_map[design_point_enum_ID] = NULL;
01264               
01265               // and then delete the nodes. it is assumed that if the local elem was created, 
01266               // the nodes too will be present, and need to be deleted
01267               for (unsigned int i = 0; i< n_nodes; i++)
01268                 delete this->local_elem_nodes_for_DV_map[design_point_enum_ID][i];
01269               
01270               // finally, delete the array that was created to store the nodes.
01271               delete[] this->local_elem_nodes_for_DV_map[design_point_enum_ID];
01272               this->local_elem_nodes_for_DV_map[design_point_enum_ID] = NULL;
01273             }
01274             break;
01275           }
01276       }
01277     }
01278   
01279   // next, set the initialized bools to false
01280   this->local_elem_is_initialized_for_DV_map[design_point_enum_ID] = false;
01281 }
