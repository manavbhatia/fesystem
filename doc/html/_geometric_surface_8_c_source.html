00026 bool GeometricSurface::checkIfDuplicate(const GeometricEntity& surf_ref)
00027 {
00028   // make sure that the two surfaces are not the same object
00029   assert (&surf_ref != this);
00030 
00031   const unsigned int n_lines = this->lines.size();
00032 
00033   // also, its type should be a surface
00034   if (surf_ref.type() != GeometricEntity::SURFACE)
00035     return false;
00036 
00037   // convert this entity's reference to a geometric surface
00038   const GeometricSurface& surf = dynamic_cast<const GeometricSurface&>(surf_ref);
00039 
00040   // get the lines for the given surface
00041   const std::vector<LineAndOrientationPair>& surf_lines = surf.getLines();
00042   if (n_lines != surf_lines.size())
00043     return false;
00044 
00045   // get the line IDs for the two surfaces, and take into account the orientation.
00046   // the tests for now will be simple enough where it is assumed that for the two surfaces, if the 
00047   // bounding lines are the same, they are duplicate. Then, if that is the case, then the orientation
00048   // of one of the lines in bot surfaces will give the orientation of the surface. i.e., if line 1 has ID 45 
00049   // in surface 1 and -45 in surface 2, then the two orientations are the opposite in direction.
00050   
00051   unsigned int id_sum=0;
00052 
00053   for (unsigned int i=0; i < n_lines; i++)
00054     for (unsigned int j=0; j < n_lines; j++)
00055       {
00056         if (this->lines[i].first->baseEntityID() == surf_lines[j].first->baseEntityID())
00057           id_sum++;
00058       }
00059 
00060   if (id_sum > n_lines)
00061     abort(); // a surface cannot have multiple lines by same ID
00062   else if (id_sum < n_lines)
00063     return false; // the two surfaces do not share the same boundary
00064   else 
00065     {
00066       // this implies that id_sum == n_lines
00067       
00068       // now set the base entity ID and return
00069       this->is_duplicate = true;
00070       this->base_entity_ID = surf.baseEntityID();
00071 
00072       // check the orientation of any one line to establish the orientation of this surface
00073       // with respect to the given surface
00074       for (unsigned int j=0; j < n_lines; j++)
00075         if (this->lines[0].first->baseEntityID() == surf_lines[j].first->baseEntityID())
00076           {
00077             if (this->lines[0].second == surf_lines[j].second)
00078               this->same_orientation_as_base_entity = true;
00079             else 
00080               this->same_orientation_as_base_entity = false;
00081 
00082             break;
00083           }
00084 
00085       return false;
