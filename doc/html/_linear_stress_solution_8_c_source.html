00055 Solution::LinearStressSolution::solve()
00056 {
00057   // if a coupled thermal-structural response is requested, then
00058   // first perform a thermal analysis, import the loads to structural and
00059   // then perform the structural solution
00060   if (this->coupled_thermoelastic)
00061     {
00062     const Discipline::DisciplineInfo& thermal_info = 
00063     this->fesystem_controller->analysis_case->getAnalysisDisciplineInfo
00064     (Discipline::THERMAL_DISCIPLINE::num());
00065     
00066     switch (thermal_info.getAnalysisTypeEnumID())
00067       {
00068       case LINEAR_ANALYSIS_ENUM_ID:
00069         this->linearSolve(Discipline::THERMAL_DISCIPLINE::num());
00070         break;
00071         
00072       case NONLINEAR_ANALYSIS_ENUM_ID:
00073         this->nonlinearSolve(Discipline::THERMAL_DISCIPLINE::num());
00074         break;
00075         
00076       default:
00077         Assert(false, ExcInternalError());
00078         break;
00079       }
00080     
00081     // get the interpolation case from the analysis case
00082     const InterpolationCase& interpolation_case = 
00083       this->fesystem_controller->analysis_case->getInterpolationCase(this->interpolation_case_ID);
00084     std::auto_ptr<DirectInterpolation> interpolation
00085       (new DirectInterpolation(*(this->fesystem_controller), interpolation_case));
00086     interpolation->interpolate();
00087     }
00088   
00089   this->linearSolve(Discipline::STRUCTURAL_DISCIPLINE::num());
00090 }
