00281 FEInterpolationElem::calculate_F(DenseVector<double>* f_vector, 
00282                                  const unsigned int design_point_enum_ID,
00283                                  bool sensitivity_calculation)
00284 {
00285   // calculation of the element load will involve the following procedure
00286   // -- create a vector of global coordinates of the Gauss points 
00287   //    of this element.
00288   // -- find out the element on the 'from_mesh' inside which this element
00289   // belongs.
00290   // -- find out the value of the dof to be interpolated at the point
00291   // -- once this has been done for all gauss points, caluculate the force
00292   //    vector
00293   
00294   // initialize the element if it is not already initialized
00295   if (this->local_elem_is_initialized_for_DV_map[design_point_enum_ID] == false)
00296     this->initialize_element(design_point_enum_ID);
00297   
00298   // get the phi for this elem, and the nodes for the base elem
00299   // multiply the phi and node location and get the localtion of the
00300   // element
00301         
00302   const std::vector<std::vector<Real> >& phi = 
00303     this->fe_base_map_for_DV[design_point_enum_ID][LAGRANGE]->get_phi();
00304   const std::vector<Real>& JxW =
00305     this->fe_base_map_for_DV[design_point_enum_ID][LAGRANGE]->get_JxW();
00306   
00307   
00308   std::vector<Point> gauss_coords(phi[0].size());
00309   
00310   unsigned int n_quad_points = phi[0].size();
00311   unsigned int n_nodes = phi.size(); 
00312   
00313   double x=0.0, y=0.0, z=0.0;
00314   std::vector<Elem*> source_elems;
00315   unsigned int mesh_ID = this->analysis_discipline.getAnalysisMeshID();
00316   
00317   FEInterpolation* fe_interpolation = 
00318     dynamic_cast<FEInterpolation*> (&(this->analysis_discipline.getAnalysisDriver()));
00319   
00320   for (unsigned int j=0; j<n_quad_points; j++) // iterate over each gauss point
00321     {
00322     x = 0.0, y=0.0, z=0.0;
00323     for (unsigned int i=0; i< n_nodes; i++)  // iterate over each node shape function
00324       {
00325       gauss_coords[j].add_scaled(this->geometric_elems_for_DV_map[design_point_enum_ID]->point(i), phi[i][j]);
00326       }
00327     
00328     // for each point, ask the interpolation discipline to find the element 
00329     // which contains the point
00330     source_elems.push_back(fe_interpolation->getElemContainingPoint
00331                            (gauss_coords[j],
00332                             this->elem_ID,
00333                             mesh_ID));
00334     }
00335   
00336   
00337   // now, for each quadrature point, get the local coordinates in the source
00338   // elem, calculate the value of the variable at the point and add it to the 
00339   // force vector
00340   Point inverse_coords;
00341   std::auto_ptr<FEBase> fe(NULL);
00342   
00343   switch (this->dimension)
00344     {
00345     case 1:
00346       {
00347         fe.reset(new FE<1,LAGRANGE>(FEType()));
00348       }
00349       break;
00350       
00351     case 2:
00352       {
00353         fe.reset(new FE<2,LAGRANGE>(FEType()));
00354       }
00355       break;
00356       
00357     case 3:
00358       {
00359         fe.reset(new FE<3,LAGRANGE>(FEType()));
00360       }
00361       break;
00362       
00363     default:
00364       abort();
00365     }
00366   
00367   DenseVector<double> source_dof_values, qp_values(n_quad_points);
00368   std::vector<Point> point_vec;
00369   
00370   for (unsigned int qp =0; qp < n_quad_points; qp++)
00371     {
00372     // get the shape function values in the source elem
00373     // this has to be initialized as a 2-D element. This is because libMesh needs the 
00374     // elements of n dim to live in an n-dim space for the method inverse map.
00375     // this needs to be looked into in greater detail to solve the problem.
00376     inverse_coords = FE<2,LAGRANGE>::inverse_map(source_elems[qp], 
00377                                                  gauss_coords[qp]);
00378     // calculate the value of the source variable at this 
00379     // point
00380     if (source_dof_values.size() != (source_elems[qp])->n_nodes())
00381       source_dof_values.resize((source_elems[qp])->n_nodes());
00382     source_dof_values.zero();
00383     
00384     fe_interpolation->getDofValuesForSourceElem(source_elems[qp],
00385                                                 source_dof_values);
00386     point_vec.clear();
00387     point_vec.push_back(inverse_coords);
00388     fe->reinit(source_elems[qp], &point_vec);
00389     
00390     const std::vector<std::vector<Real> >& source_phi = fe->get_phi();
00391     for (unsigned int i=0; i < source_phi.size(); i++)
00392       qp_values(qp) += source_phi[i][0] * source_dof_values(i);
00393     }
00394   
00395   
00396   
00397   for (unsigned int qp=0; qp < n_quad_points; qp++)
00398     for (unsigned int i=0; i < phi.size(); i++)
00399       (*f_vector)(i) += JxW[qp] * phi[i][qp] * qp_values(qp);
00400   
00401 }
