00072 inline void
00073 NameEnumerationHandler::registerNameAndID(const unsigned int ID, 
00074                                           const std::string& name)
00075 {
00076   // find the name in the map. If it does not exist, add it, else return the 
00077   // id for this name
00078   NameEnumerationHandler::NameToIDMap::const_iterator name_it, name_end;
00079   name_it = this->name_to_id_map.find(name);
00080   name_end = this->name_to_id_map.end();
00081   
00082   NameEnumerationHandler::IDToNameMap::const_iterator id_it, id_end;
00083   id_it = this->id_to_name_map.find(ID);
00084   id_end = this->id_to_name_map.end();
00085   
00086   // make sure that the id and name are unique
00087   Assert(name_it == name_end, ExcFESystemNameAlreadyExists(ID, name));
00088   Assert(id_it == id_end, ExcFESystemIDAlreadyExists(ID, name));
00089   
00090   
00091   // now add the name and ID to the map
00092   std::pair<NameEnumerationHandler::NameToIDMap::iterator, bool> 
00093     return_pair = this->name_to_id_map.insert
00094     (NameEnumerationHandler::NameToIDMap::value_type(name, ID));
00095   
00096   assert (return_pair.second == true); 
00097   
00098   std::pair<NameEnumerationHandler::IDToNameMap::iterator, bool>
00099     reverse_map_return_pair = this->id_to_name_map.insert
00100     (NameEnumerationHandler::IDToNameMap::value_type(ID, name));
00101   
00102   assert (return_pair.second == true);    
00103 }
