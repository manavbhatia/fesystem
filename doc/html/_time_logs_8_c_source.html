00179                                     const std::string& header)
00180 {
00181   if (this->event_map.count(header) == 0)
00182     this->event_map[header] = new HeaderEventData;
00183   
00184   FESystemUtility::HeaderEventData& header_data = *(this->event_map[header]);
00185   
00186   switch (header_data.getCurrentStatus())
00187     {
00188     case FESystemUtility::HeaderEventData::RUNNING:
00189       {
00190         if (header_data.getCurrentEvent() == event_name)

00193           {
00194           this->performance_logging->pause_event(header_data.getCurrentEvent(),
00195                                                  header);

00197           
00198           // if the event has been called multiple times, it will be restarted, instead 
00199           // of being started, since that means that the event was paused earlier
00200           if (header_data.getEventCallCount(header_data.getCurrentEvent()) > 1)
00201             this->performance_logging->restart_event(header_data.getCurrentEvent(),
00202                                                      header);
00203           else
00204             this->performance_logging->start_event(header_data.getCurrentEvent(),
00205                                                    header);
00206           
00207           }
00208       }      
00209       break;
00210       
00211     case FESystemUtility::HeaderEventData::STOPPED:
00212       {
00213         // there is no logging being performed, and hence, the first event 
00214         // is being started now.
00215         header_data.setCurrentEvent(event_name);

00233   Assert (header_data.getCurrentEvent() == event_name, ExcInternalError());

00237   // if the next event is also the same as the event just stopped, then do not stop logging
00238   if (header_data.eventsExist())

00240     // even after removing this event, if the call count is still zero for it, 
00241     // then pause the event, and do not stop it

00243       this->performance_logging->pause_event(event_name, header);
00244     else
00245       this->performance_logging->stop_event(event_name, header);
00246     
00247     // next, simply restart the current event.
00248     this->performance_logging->restart_event(header_data.getCurrentEvent(), header);
