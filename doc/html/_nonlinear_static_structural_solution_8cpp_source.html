00068 {
00069   // if a coupled thermal-structural response is requested, then
00070   // first perform a thermal analysis, import the loads to structural and
00071   // then perform the structural solution
00072   if (this->coupled_thermoelastic)
00073     {
00074       const Discipline::DisciplineInfo& thermal_info = 
00075       this->fesystem_controller->analysis_case->getAnalysisDisciplineInfo
00076       (Discipline::THERMAL_DISCIPLINE::num());
00077       
00078       switch (thermal_info.getAnalysisTypeEnumID())
00079       {
00080         case LINEAR_ANALYSIS_ENUM_ID:
00081           this->linearSolve(Discipline::THERMAL_DISCIPLINE::num());
00082           break;
00083           
00084         case NONLINEAR_ANALYSIS_ENUM_ID:
00085           this->nonlinearSolve(Discipline::THERMAL_DISCIPLINE::num());
00086           break;
00087           
00088         default:
00089           Assert(false, ExcInternalError());
00090           break;
00091       }
00092       
00093       // get the interpolation case from the analysis case
00094       const InterpolationCase& interpolation_case = 
00095       this->fesystem_controller->analysis_case->getInterpolationCase(this->interpolation_case_ID);
00096       std::auto_ptr<DirectInterpolation> interpolation
00097       (new DirectInterpolation(*(this->fesystem_controller), interpolation_case));
00098       interpolation->interpolate();
00099     }
00100   
00101   this->nonlinearTransientSolve(Discipline::STRUCTURAL_DISCIPLINE::num());
00102 }
