        // also insert this elem into the entity ID to elem map
        switch (this_elem->type())
          {
          case EDGE2:

              this->line_entity_ID_elem_map.insert(std::multimap<unsigned int, Elem*>::value_type(reg_elem, this_elem));

            break;
          }
