00284 FESystemElem::Bar::calculate_F_T(DenseVector<double>* vector, 
00285                                 const unsigned int design_point_enum_ID,
00286                                 bool sensitivity_calculation)
00287 {
00288   // param not used here
00289   (void) design_point_enum_ID;
00290 
00291   // the load vector is calculated based on the following expression
00292   // 
00293   //                                    | -1    -1      | |T1|                                  |1 |
00294   // F_T =  (E*A*alpha/2.0)*    |               | |   | + (E*A*alpha* T_ref)    |  |
00295   //                                    | 1  1          | |T2|                                  |-1|
00296   //
00297   //
00298   
00299   static double ref_temp = this->analysis_discipline.getFESystemController().
00300   analysis_case->getRealParameter("REFERENCE_TEMP");
00301   
00302   static unsigned int n_nodes;
00303   n_nodes = this->getNNodes();
00304   // initialize the data structures for the load
00305   static DenseVector<double> nodal_temp(n_nodes), nodal_temp_sens(n_nodes);
00306   
00307   if (nodal_temp.size() != n_nodes)
00308     {
00309     nodal_temp.resize(n_nodes);
00310     nodal_temp_sens.resize(n_nodes);
00311     }
00312   
00313   nodal_temp.zero(); nodal_temp_sens.zero();
00314   
00315   this->extractNodalTemperatureVectorFromLoads(nodal_temp, false);
00316   
00317   static double factor = 0.0, factor_sens = 0.0, factor_temp_sens = 0.0;
00318   factor = 0.0; factor_sens = 0.0; factor_temp_sens = 0.0;
00319   this->elem_property_card->getFactor(factor, THERMAL_EXPANSION_FACTOR::num());
00320   
00321   if (sensitivity_calculation)
00322     {
00323     // get the nodal temperature vector sensitivity
00324     this->extractNodalTemperatureVectorFromLoads(nodal_temp_sens, true);
00325     
00326     switch (this->sensitivity_parameter)
00327       {
00328       case PROPERTY_PARAMETER_ENUM_ID:
00329         this->elem_property_card->getFactorSensitivityForGlobalParameter
00330         (factor_sens, THERMAL_EXPANSION_FACTOR::num(), this->sensitivity_parameter_ID);
00331         
00332       default:
00333         if (this->analysis_discipline.checkPropertyDependenceOnTemperature())
00334           this->elem_property_card->getFactorSensitivityForLocalParameter
00335             (factor_temp_sens, THERMAL_EXPANSION_FACTOR::num(), Property::TEMPERATURE::num());
00336         break;
00337       }
00338     }
00339   
00340   // now, scale the matrix with the right factors to take into 
00341   // account the sensitivity if needed
00342   if (sensitivity_calculation)
00343     {
00344     // this load vector does not have any shape effect, hence, the
00345     // shape and property sensitivity are both similar, where 
00346     // the factor_sens will be zero for shape sensitivity
00347     // also, the reference temperature is assumed to be constant
00348     (*vector)(0) = 
00349     - (factor_sens + factor_temp_sens * 0.5 * (nodal_temp_sens(0) + nodal_temp_sens(1))) * 
00350     0.5 * (nodal_temp(0) + nodal_temp(1) - 2.0 * ref_temp) +
00351     factor * 0.5 * (nodal_temp_sens(0) + nodal_temp_sens(1));
00352     (*vector)(1) = -(*vector)(0);
00353     }
00354   else
00355     {
00356     (*vector)(0) = - factor * 0.5 * (nodal_temp(0) + nodal_temp(1)) + factor * ref_temp; // u1
00357     (*vector)(1) =  -(*vector)(0); // u2
00358     }
00359 }
