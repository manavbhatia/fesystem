00017                  const double &s2,
00018                  const double &x)

00031                  const double &s1,
00032                  const double &s2,
00033                  const double &x)       
00034 {
00035   double b,f;
00036   b = sqrt(4.0 * s0 * s2 - s1 * s1);
00037   f = 2.0 * (-s1) * b * b * atan2((s1 + 2.0 * s2 * x), b);
00038   f += b * (2.0 * s2 * x * (s1 - s2 * x) + 
00039             (2.0 * s2 * (s2 * x * x + s0) - s1 * s1) * log(s0 + x * (s1 + s2 * x)));
00040   f /= (4.0 * s2 * s2 * b);
