00321   // first of all, clear the map
00322   this->elem_vec_map.clear();
00323   
00324   // next, iterate over all the elems, and based on the elem type, 
00325   // insert them in the vector. Finally, insert the vectors in the 
00326   // map
00327   // get the iterator for the elems
00328   MeshBase::const_element_iterator elem_it = this->mesh->elements_begin();
00329   MeshBase::const_element_iterator elem_end = this->mesh->elements_end();
00330   
00331   std::set<Elem*>::const_iterator dummy_elem_end =
00332   this->dummy_elem_set->end();
00333   
00334   std::vector<Elem*> *elem_vec=NULL;
00335         
00336   for (; elem_it != elem_end; elem_it++)
00337     {
00338       if (this->dummy_elem_set->find(*elem_it) != dummy_elem_end)
00339         continue;
00340       
00341       
00342       // insert this elem type in the map if it does not already exist
00343       if (this->elem_vec_map.count((**elem_it).type()) == 0)
00344         {
00345           elem_vec = 
00346           &(this->elem_vec_map.insert(std::map<ElemType, std::vector<Elem*> >::value_type
00347                                       ((**elem_it).type(), std::vector<Elem*>())).first->second);
00348         }
00349       else
00350         elem_vec = &(this->elem_vec_map.find((**elem_it).type())->second);
00351       
00352       elem_vec->push_back(*elem_it);
00353     }
