00336   // class not handled

00540 Solver::SlepcEigenSolver::getEigenValue(unsigned int eigen_index,
00541                                    double* real_value,
00542                                    double* img_value)
00543 {
00544   // make sure that the requested index is less than the number of converged 
00545   // eigenpairs
00546   Assert(this->solution_completed, ExcInvalidState());
00547   Assert(this->n_converged_eigen_pairs > 0, ExcInternalError());
00548   Assert((int) eigen_index < this->n_converged_eigen_pairs, ExcInternalError());
00549   
00550   // now get the eigen pair
00551   PetscErrorCode ierr = 0;
00552   if (this->eigen_solver_info.ifSwapMatrices())
00553     {
00554     double rval=0.0, ival=0.0, mag=0.0;
00555     ierr = EPSGetValue(this->eps, this->eigen_val_permutation[eigen_index],
00556                        &rval, &ival);
00557     CHKERRABORT(FESystem::COMM_WORLD, ierr);
00558     mag = sqrt(rval * rval + ival * ival);
00559     *real_value = rval / mag;
00560     *img_value = ival / mag;
00561     }
00562   else
00563     {
00564     ierr = EPSGetValue(this->eps, this->eigen_val_permutation[eigen_index],
00565                        real_value, img_value);
