00224   // get the stress and mechanical strain. 
00225   
00226   if (DV_ID == 0)

00228       TensorValue<double>& stress = this->getStressTensor(load_case);

00231       // multiply the two to get the strain energy
00232       double strain_energy_density = 0.0;
00233       for (unsigned int i=0; i<2; i++)
00234         for (unsigned int j=0; j<2; j++) 
00235           strain_energy_density += stress(i,j)*strain(i,j) ; 
00236       return strain_energy_density * 0.5;
00237     }
00238   else 
00239     {
00240       // get the sensitivity and strain
00241       TensorValue<double>& stress = this->getStressTensor(load_case);
00242       TensorValue<double>& strain = this->getMechanicalStrainTensor(load_case);
00243       TensorValue<double>& stress_sens = this->getStressTensor(load_case, DV_ID);
00244       TensorValue<double>& strain_sens = this->getMechanicalStrainTensor(load_case, DV_ID);
00245 
00246       double strain_energy_density_sens = 0.0;
00247       for (unsigned int i=0; i<2; i++)
00248         for (unsigned int j=0; j<2; j++) 
00249           strain_energy_density_sens += stress_sens(i,j)*strain(i,j) +
00250             stress(i,j) * strain_sens(i,j);
