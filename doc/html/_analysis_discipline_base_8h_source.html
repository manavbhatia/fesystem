00675                                                     FESystemElem::FESystemElemBase*& fesys_elem)

00680   
00681   // if it does not exist, create and insert the elem
00682   if (elem_exists == 0)
00683     {

00688       this->fesystem_elem_map.insert(Discipline::AnalysisDisciplineBase::FESystemElemMap::
00689                                      value_type(elem_kind_enum_ID, local_fesys_elem)).second;
