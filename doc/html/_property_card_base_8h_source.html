00297 Property::PropertyCardBase::
00298 reinitLocalParameters(const std::map<unsigned int, double>* value_map)
00299 {
00300   // before calling this function, the user should have cleared previous initializations
00301   Assert(this->local_params_initialized == false,
00302          ExcInvalidState());
00303   
00304   
00305   static std::vector<unsigned int> reinitialized_param_ids;
00306   reinitialized_param_ids.clear();
00307   
00308   // set the values of the parameters, and calculate and store the function values 
00309   // and their derivatives. 
00310   // a list of all function values that are set to their default values will have be 
00311   // stored so that there is no need to initialize them or clear them when a partial 
00312   // clear is performed 
00313   Property::PropertyCardBase::LocalParameterMap::iterator 
00314   param_it, param_end;
00315   param_it = this->local_parameter_map.begin();
00316   param_end = this->local_parameter_map.end();
00317   
00318   double value = 0.0;
00319   
00320   bool insert_success = false;
00321   // if the map is null, then all params will have their default values
00322   if (value_map == NULL)
00323     {
00324     for (; param_it != param_end; param_it++)
00325       {
00326       // if the local parameter has not already been initialized to its local value, 
00327       // initialize it
00328       if (this->default_local_params.count(param_it->second->getID()) == 0)
00329         {
00330         value = param_it->second->getReferenceValue();
00331         param_it->second->setCurrentValue(value);
00332         
00333         // also add this param to the set of params that have been set to their default 
00334         // values.
00335         insert_success = this->default_local_params.insert(param_it->second->getID()).second;
00336         Assert(insert_success, FESystemExceptions::ExcDuplicateID("Local Parameter", 
00337                                                                   param_it->second->getID()));
00338         
00339         reinitialized_param_ids.push_back(param_it->second->getID());
00340         }
00341       }
00342     }
00343   else
00344     {
00345     
00346     // now set the values of the parameters by syncing the parameters that have been 
00347     // specified in the input
00348     unsigned int param_count_in_local_default_param_map = 0, 
00349     param_enum_id = 0;
00350     
00351     std::map<unsigned int, double>::const_iterator input_param_it, input_param_end;
00352     input_param_end = value_map->end();
00353     
00354     for (; param_it != param_end; param_it++)
00355       {
00356       param_enum_id = param_it->first;
00357       input_param_it = value_map->find(param_enum_id);
00358       
00359       // check if this has been set to its default value
00360       param_count_in_local_default_param_map = 
00361         this->default_local_params.count(param_it->second->getID());
00362       
00363       if (input_param_it == input_param_end)
00364         {
00365         // if this param has not already been initialized to its default value, 
00366         // then do the same, and add this ID to the set.
00367         // otherwise, if it has already been initialized to its local value, then
00368         // no need to do anything
00369         if (param_count_in_local_default_param_map == 0)
00370           {
00371           value = param_it->second->getReferenceValue();
00372 
00373           // also add this param to the set of params that have been set to their default 
00374           // values.
00375           insert_success = this->default_local_params.insert(param_it->second->getID()).second;
00376           Assert(insert_success, FESystemExceptions::ExcDuplicateID("Local Parameter", 
00377                                                                     param_it->second->getID()));
00378           }
00379         else
00380           continue;
00381         }
00382       else
00383         {
00384         // we need to set the value of this param to the one specified in the input. 
00385         // for this, we need to check if the param has been previously set to its 
00386         // default value. If it has been, then its ID needs to be cleared from the 
00387         // local default param set
00388         if (param_count_in_local_default_param_map != 0)
00389           this->default_local_params.erase(param_it->second->getID());
00390         
00391         value = input_param_it->second;
00392         }
00393       
00394       param_it->second->setCurrentValue(value);
00395       reinitialized_param_ids.push_back(param_it->second->getID());
00396       }
00397     
00398     }
00399   
00400   
00401   // once the values have been set, the functions need to be initialized
00402   // call the function value initialization for the parameter IDs
00403   // that have just been initialized
00404   this->reinitParameterFunctionValues(reinitialized_param_ids);
00405   
00406   // finally, set the tag for initialization of the local params
00407   this->local_params_initialized = true;
00408 }
