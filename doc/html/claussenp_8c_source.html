00134 /*
00135  * evaluate a chebyshev series
00136  * adapted from fortran csevl
00137  */
00138 real
00139 csevl( real x, real *cs, int n )
00140 {
00141   real b2, b1 = 0, b0 = 0, twox = 2 * x;
00142 
00143   while( n-- ){
00144     b2 = b1;
00145     b1 = b0;
00146     b0 = twox * b1 - b2 + cs[n];
00147   }
00148 
00149   return HALF * ( b0 - b2 );
00150 }

00152 /*
00153  * from the original fortran inits
00154  * april 1977 version.  w. fullerton, c3, los alamos scientific lab.
00155  *
00156  * initialize the orthogonal series so that inits is the number of terms
00157  * needed to insure the error is no larger than eta.  ordinarily, eta
00158  * will be chosen to be one-tenth machine precision.
00159  */
00160 static int
00161 inits( real *series, int n, real eta )
00162 {
00163   real err = 0;
00164 
00165   while( err <= eta && n-- ){
00166     err += fabs( series[n] );
00167   }
00168 
00169   return n++;
00170 }

00173 claussen( real x )
00174 {
00175   static int nclpi6 = 0, nclpi2 = 0, ncl5pi6 = 0;
00176   /*
00177    * right half (Pi <= x < 2 Pi)
00178    */
00179   int rh = 0;
00180   real f;
00181 
00182   if( !nclpi6 ){
00183     nclpi6 = inits( clpi6, sizeof clpi6 / sizeof *clpi6, mach[2] / 10 );
00184     nclpi2 = inits( clpi2, sizeof clpi2 / sizeof *clpi2, mach[2] / 10 );
00185     ncl5pi6 = inits( cl5pi6, sizeof cl5pi6 / sizeof *cl5pi6, mach[2] / 10 );
00186   }
00187 
00188   /*
00189    * get to canonical interval
00190    */
00191   if( ( x = fmod( x, 2 * M_PI ) ) < 0 ){
00192     x += ( real )( 2 * M_PI );
00193   }
00194   if( x > ( real )M_PI ){
00195     rh = 1;
00196     x = ( real )( 2 * M_PI ) - x;
00197   }
00198 
00199   if( x == 0 ){
00200     f = x;
00201   }else if( x <= ( real )( M_PI / 3 ) ){
00202     f = csevl( x * ( real )( 6 / M_PI ) - 1, clpi6, nclpi6 ) * x
00203       - x * log( x );
00204   }else if( x <= ( real )( 2 * M_PI / 3 ) ){
00205     f = csevl( x * ( real )( 3 / M_PI ) - 1, clpi2, nclpi2 ) * x
00206       - x * log( x );
00207   }else{ /* x <= Pi */
00208     f = ( ( real )M_LN2 -
00209          csevl( 5 - x * ( real )( 6 / M_PI ), cl5pi6, ncl5pi6 ) ) *
00210            ( ( real )M_PI - x );
00211   }
00212 
00213   return rh ? -f : f;
00214 }
