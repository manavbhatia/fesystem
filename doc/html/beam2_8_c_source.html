00414 FESystemElem::Beam2::calculate_F_T(DenseVector<double>* vector, 
00415                                    const unsigned int design_point_enum_ID,
00416                                    bool sensitivity_calculation)
00417 {
00418   // param not used here

00420 
00421   // the load vector is calculated based on the following expression
00422   // 
00423   //                                    | -1    -1      | |T1|                                  |1 |
00424   // F_T =  (E*A*alpha/2.0)*    |               | |   | + (E*A*alpha* T_ref)    |  |
00425   //                                    | 1  1          | |T2|                                  |-1|
00426   //
00427   //
00428

00430   analysis_case->getRealParameter("REFERENCE_TEMP");
00431   
00432   static unsigned int n_nodes;
00433   n_nodes = this->getNNodes();
00434   // initialize the data structures for the load
00435   static DenseVector<double> nodal_temp(n_nodes), nodal_temp_sens(n_nodes);
00436 
00437   if (nodal_temp.size() != n_nodes)
00438     {
00439     nodal_temp.resize(n_nodes);
00440     nodal_temp_sens.resize(n_nodes);
00441     }
00442   
00443   nodal_temp.zero(); nodal_temp_sens.zero();
00444   
00445   this->extractNodalTemperatureVectorFromLoads(nodal_temp, false);
00446   
00447   static double factor = 0.0, factor_sens = 0.0, factor_temp_sens = 0.0;
00448   factor = 0.0; factor_sens = 0.0; factor_temp_sens = 0.0;
00449   this->elem_property_card->getFactor(factor, THERMAL_EXPANSION_FACTOR::num());
00450   
00451   if (sensitivity_calculation)
00452     {
00453     // get the nodal temperature vector sensitivity
00454     this->extractNodalTemperatureVectorFromLoads(nodal_temp_sens, true);
00455     
00456     switch (this->sensitivity_parameter)
00457       {
00458       case PROPERTY_PARAMETER_ENUM_ID:
00459         this->elem_property_card->getFactorSensitivityForGlobalParameter
00460         (factor_sens, THERMAL_EXPANSION_FACTOR::num(), this->sensitivity_parameter_ID);
00461         
00462       default:
00463         if (this->analysis_discipline.checkPropertyDependenceOnTemperature())
00464           this->elem_property_card->getFactorSensitivityForLocalParameter
00465             (factor_temp_sens, THERMAL_EXPANSION_FACTOR::num(), Property::TEMPERATURE::num());
00466         break;
00467       }
00468     }
00469   
00470   // now, scale the matrix with the right factors to take into 
00471   // account the sensitivity if needed
00472   if (sensitivity_calculation)
00473     {
00474     // this load vector does not have any shape effect, hence, the
00475     // shape and property sensitivity are both similar, where 
00476     // the factor_sens will be zero for shape sensitivity

00483     }

00485     {
